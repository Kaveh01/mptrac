68  lat1 = scan_ctl(argv[1], argc, argv, "SPLIT_LAT1", -1, "0", NULL);

102  kmin = gsl_stats_min(kk, 1, (size_t) nz);
103  for (iz = 0; iz < nz; iz++)

109  for (ip = 0; ip < atm->np; ip++) {

111  mmax = GSL_MAX(mmax, atm->q[ctl.qnt_m][ip]);

117  for (i = 0; i < n; i++) {

121  do {
122  ip = (int) gsl_rng_uniform_int(rng, (long unsigned int) atm->np);
123  } while (gsl_rng_uniform(rng) > atm->q[ctl.qnt_m][ip] / mmax);
124  else

126 
127  /* Set time... */
128  if (t1 > t0)
129  atm2->time[atm2->np] = t0 + (t1 - t0) * gsl_rng_uniform_pos(rng);
130  else

132  + gsl_ran_gaussian_ziggurat(rng, dt / 2.3548);
133 
134  /* Set vertical position... */
135  do {
136  if (nz > 0) {
137  do {
138  z = zmin + (zmax - zmin) * gsl_rng_uniform_pos(rng);
139  iz = locate_irr(kz, nz, z);
140  k = LIN(kz[iz], kk[iz], kz[iz + 1], kk[iz + 1], z);
141  } while (gsl_rng_uniform(rng) > k);
142  atm2->p[atm2->np] = P(z);
143  } else if (z1 > z0)
144  atm2->p[atm2->np] = P(z0 + (z1 - z0) * gsl_rng_uniform_pos(rng));
145  else
146  atm2->p[atm2->np] = atm->p[ip]
147  + DZ2DP(gsl_ran_gaussian_ziggurat(rng, dz / 2.3548), atm->p[ip]);
148  } while (atm2->p[atm2->np] < P(100.) || atm2->p[atm2->np] > P(-1.));
149 
150  /* Set horizontal position... */
151  if (lon1 > lon0 && lat1 > lat0) {
152  atm2->lon[atm2->np] = lon0 + (lon1 - lon0) * gsl_rng_uniform_pos(rng);
153  atm2->lat[atm2->np] = lat0 + (lat1 - lat0) * gsl_rng_uniform_pos(rng);
154  } else {
155  atm2->lon[atm2->np] = atm->lon[ip]
156  + gsl_ran_gaussian_ziggurat(rng, DX2DEG(dx, atm->lat[ip]) / 2.3548);
157  atm2->lat[atm2->np] = atm->lat[ip]
158  + gsl_ran_gaussian_ziggurat(rng, DY2DEG(dx) / 2.3548);
159  }
160 
161  /* Copy quantities... */
162  for (iq = 0; iq < ctl.nq; iq++)
163  atm2->q[iq][atm2->np] = atm->q[iq][ip];
164 
165  /* Adjust mass... */
166  if (ctl.qnt_m >= 0)
167  atm2->q[ctl.qnt_m][atm2->np] = mtot / n;
168 
169  /* Increment particle counter... */
170  if ((++atm2->np) > NP)
171  ERRMSG("Too many air parcels!");
172  }
173 
174  /* Save data and close file... */
175  write_atm(argv[3], &ctl, atm2, 0);
176 
177  /* Free... */
178  free(atm);
179  free(atm2);
180 
181  return EXIT_SUCCESS;
