47 int main(
48  int argc,
49  char *argv[]) {
50 
51  ctl_t ctl;
52 
53  met_t *met;
54 
55  FILE *out;
56 
57  static double cutImag[PMAX], cutReal[PMAX], lx[PMAX], A[PMAX], phi[PMAX],
58  wavemax;
59 
60  /* Allocate... */
61  ALLOC(met, met_t, 1);
62 
63  /* Check arguments... */
64  if (argc < 4)
65  ERRMSG("Give parameters: <ctl> <spec.tab> <met0>");
66 
67  /* Read control parameters... */
68  read_ctl(argv[1], argc, argv, &ctl);
69  wavemax =
70  (int) scan_ctl(argv[1], argc, argv, "SPEC_WAVEMAX", -1, "7", NULL);
71 
72  /* Read meteorological data... */
73  if (!read_met(&ctl, argv[3], met))
74  ERRMSG("Cannot read meteo data!");
75 
76  /* Create output file... */
77  printf("Write spectral data file: %s\n", argv[2]);
78  if (!(out = fopen(argv[2], "w")))
79  ERRMSG("Cannot create file!");
80 
81  /* Write header... */
82  fprintf(out,
83  "# $1 = time [s]\n"
84  "# $2 = altitude [km]\n"
85  "# $3 = longitude [deg]\n" "# $4 = latitude [deg]\n");
86  for (int ix = 0; ix <= wavemax; ix++) {
87  fprintf(out, "# $%d = wavelength (PW%d) [km]\n", 5 + 3 * ix, ix);
88  fprintf(out, "# $%d = amplitude (PW%d) [K]\n", 6 + 3 * ix, ix);
89  fprintf(out, "# $%d = phase (PW%d) [deg]\n", 7 + 3 * ix, ix);
90  }
91 
92  /* Loop over pressure levels... */
93  for (int ip = 0; ip < met->np; ip++) {
94 
95  /* Write output... */
96  fprintf(out, "\n");
97 
98  /* Loop over latitudes... */
99  for (int iy = 0; iy < met->ny; iy++) {
100 
101  /* Copy data... */
102  for (int ix = 0; ix < met->nx; ix++) {
103  cutReal[ix] = met->t[ix][iy][ip];
104  cutImag[ix] = 0.0;
105  }
106 
107  /* FFT... */
108  fft_help(cutReal, cutImag, met->nx);
109 
110  /*
111  Get wavelength, amplitude, and phase:
112  A(x) = A[0] + A[1] * cos(2 pi x / lx[1] + phi[1]) + A[2] * cos...
113  */
114  for (int ix = 0; ix < met->nx; ix++) {
115  lx[ix] = DEG2DX(met->lon[met->nx - 1] - met->lon[0], met->lat[iy])
116  / ((ix < met->nx / 2) ? (double) ix : -(double) (met->nx - ix));
117  A[ix] = (ix == 0 ? 1.0 : 2.0) / (met->nx)
118  * sqrt(gsl_pow_2(cutReal[ix]) + gsl_pow_2(cutImag[ix]));
119  phi[ix]
120  = 180. / M_PI * atan2(cutImag[ix], cutReal[ix]);
121  }
122 
123  /* Write data... */
124  fprintf(out, "%.2f %g %g %g", met->time, Z(met->p[ip]), 0.0,
125  met->lat[iy]);
126  for (int ix = 0; ix <= wavemax; ix++)
127  fprintf(out, " %g %g %g", lx[ix], A[ix], phi[ix]);
128  fprintf(out, "\n");
129  }
130  }
131 
132  /* Close file... */
133  fclose(out);
134 
135  /* Free... */
136  free(met);
137 
138  return EXIT_SUCCESS;
139 }

143 void fft_help(
144  double *fcReal,
145  double *fcImag,
146  int n) {
147 
148  gsl_fft_complex_wavetable *wavetable;
149  gsl_fft_complex_workspace *workspace;
150 
151  double data[2 * PMAX];
152 
153  int i;
154 
155  /* Check size... */
156  if (n > PMAX)
157  ERRMSG("Too many data points!");
158 
159  /* Allocate... */
160  wavetable = gsl_fft_complex_wavetable_alloc((size_t) n);
161  workspace = gsl_fft_complex_workspace_alloc((size_t) n);
162 
163  /* Set data (real, complex)... */
164  for (i = 0; i < n; i++) {
165  data[2 * i] = fcReal[i];
166  data[2 * i + 1] = fcImag[i];
167  }
168 
169  /* Calculate FFT... */
170  gsl_fft_complex_forward(data, 1, (size_t) n, wavetable, workspace);
171 
172  /* Copy data... */
173  for (i = 0; i < n; i++) {
174  fcReal[i] = data[2 * i];
175  fcImag[i] = data[2 * i + 1];
176  }
177 
178  /* Free... */
179  gsl_fft_complex_wavetable_free(wavetable);
180  gsl_fft_complex_workspace_free(workspace);
181 }
