176 
177  ctl_t ctl;

180

182

184 
185  FILE *dirlist;
186 
187  char dirname[LEN], filename[2 * LEN];
188

191  int num_devices = 0, ntask = -1, rank = 0, size = 1;
192

197 #ifdef MPI

199  MPI_Init(&argc, &argv);
200  MPI_Comm_rank(MPI_COMM_WORLD, &rank);

204  /* Initialize GPUs... */
205 #ifdef _OPENACC
206  SELECT_TIMER("ACC_INIT", NVTX_GPU);
207  num_devices = acc_get_num_devices(acc_device_nvidia);
208  if (num_devices <= 0)
209  ERRMSG("Not running on a GPU device!");
210  int device_num = rank % num_devices;
211  acc_set_device_num(device_num, acc_device_nvidia);
212  acc_device_t device_type = acc_get_device_type();

214 #endif
215 
216  /* Check arguments... */
217  if (argc < 4)
218  ERRMSG("Give parameters: <dirlist> <ctl> <atm_in>");
219 
220  /* Open directory list... */
221  if (!(dirlist = fopen(argv[1], "r")))
222  ERRMSG("Cannot open directory list!");
223 
224  /* Loop over directories... */
225  while (fscanf(dirlist, "%s", dirname) != EOF) {
226 
227  /* MPI parallelization... */
228  if ((++ntask) % size != rank)
229  continue;
230 
231  /* ------------------------------------------------------------
232  Initialize model run...
233  ------------------------------------------------------------ */
234 
235  /* Allocate... */
236  SELECT_TIMER("ALLOC", NVTX_CPU);
237  ALLOC(atm, atm_t, 1);
238  ALLOC(cache, cache_t, 1);
239  ALLOC(met0, met_t, 1);

247 #ifdef _OPENACC
248  SELECT_TIMER("CREATE_DATA_REGION", NVTX_GPU);
249 #pragma acc enter data create(atm[:1],cache[:1],ctl,met0[:1],met1[:1],dt[:NP],rs[:3*NP])
250 #endif

253  sprintf(filename, "%s/%s", dirname, argv[2]);
254  read_ctl(filename, argc, argv, &ctl);
255 
256  /* Read atmospheric data... */
257  sprintf(filename, "%s/%s", dirname, argv[3]);
258  if (!read_atm(filename, &ctl, atm))
259  ERRMSG("Cannot open file!");
260 
261  /* Set start time... */
262  SELECT_TIMER("TIMESTEPS", NVTX_CPU);
263  if (ctl.direction == 1) {
264  ctl.t_start = gsl_stats_min(atm->time, 1, (size_t) atm->np);
265  if (ctl.t_stop > 1e99)
266  ctl.t_stop = gsl_stats_max(atm->time, 1, (size_t) atm->np);
267  } else {
268  ctl.t_start = gsl_stats_max(atm->time, 1, (size_t) atm->np);
269  if (ctl.t_stop > 1e99)
270  ctl.t_stop = gsl_stats_min(atm->time, 1, (size_t) atm->np);
271  }
272 
273  /* Check time interval... */
274  if (ctl.direction * (ctl.t_stop - ctl.t_start) <= 0)
275  ERRMSG("Nothing to do! Check T_STOP and DIRECTION!");
276 
277  /* Round start time... */
278  if (ctl.direction == 1)
279  ctl.t_start = floor(ctl.t_start / ctl.dt_mod) * ctl.dt_mod;
280  else
281  ctl.t_start = ceil(ctl.t_start / ctl.dt_mod) * ctl.dt_mod;
282 
283  /* Update GPU... */
284 #ifdef _OPENACC
285  SELECT_TIMER("UPDATE_DEVICE", NVTX_H2D);
286 #pragma acc update device(atm[:1],ctl)
287 #endif
288 
289  /* Initialize random number generator... */
290  module_rng_init(ntask);
291 
292  /* Initialize meteorological data... */
293  get_met(&ctl, ctl.t_start, &met0, &met1);
294  if (ctl.dt_mod > fabs(met0->lon[1] - met0->lon[0]) * 111132. / 150.)
295  WARN("Violation of CFL criterion! Check DT_MOD!");
296 
297  /* Initialize isosurface... */
298  if (ctl.isosurf >= 1 && ctl.isosurf <= 4)
299  module_isosurf_init(&ctl, met0, met1, atm, cache);
300 
301  /* Update GPU... */
302 #ifdef _OPENACC
303  SELECT_TIMER("UPDATE_DEVICE", NVTX_H2D);
304 #pragma acc update device(cache[:1])
305 #endif
306 
307  /* ------------------------------------------------------------
308  Loop over timesteps...
309  ------------------------------------------------------------ */
310 
311  /* Loop over timesteps... */
312  for (t = ctl.t_start; ctl.direction * (t - ctl.t_stop) < ctl.dt_mod;
313  t += ctl.direction * ctl.dt_mod) {
314 
315  /* Adjust length of final time step... */
316  if (ctl.direction * (t - ctl.t_stop) > 0)
317  t = ctl.t_stop;
318 
319  /* Set time steps for air parcels... */
320  SELECT_TIMER("TIMESTEPS", NVTX_GPU);
321 #ifdef _OPENACC
322 #pragma acc parallel loop independent gang vector present(ctl,atm,atm->time,dt)
323 #endif
324  for (int ip = 0; ip < atm->np; ip++) {
325  double atmtime = atm->time[ip];
326  double tstart = ctl.t_start;
327  double tstop = ctl.t_stop;
328  int dir = ctl.direction;
329  if ((dir * (atmtime - tstart) >= 0 && dir * (atmtime - tstop) <= 0
330  && dir * (atmtime - t) < 0))
331  dt[ip] = t - atmtime;
332  else
333  dt[ip] = 0;
334  }
335 
336  /* Get meteorological data... */
337  if (t != ctl.t_start)
338  get_met(&ctl, t, &met0, &met1);
339 
340  /* Check initial positions... */
341  module_position(met0, met1, atm, dt);
342 
343  /* Advection... */
344  module_advection(met0, met1, atm, dt);
345 
346  /* Turbulent diffusion... */
347  if (ctl.turb_dx_trop > 0 || ctl.turb_dz_trop > 0
348  || ctl.turb_dx_strat > 0 || ctl.turb_dz_strat > 0)
349  module_diffusion_turb(&ctl, atm, dt, rs);
350 
351  /* Mesoscale diffusion... */
352  if (ctl.turb_mesox > 0 || ctl.turb_mesoz > 0)
353  module_diffusion_meso(&ctl, met0, met1, atm, cache, dt, rs);
354 
355  /* Convection... */
356  if (ctl.conv_cape >= 0
357  && (ctl.conv_dt <= 0 || fmod(t, ctl.conv_dt) == 0))
358  module_convection(&ctl, met0, met1, atm, dt, rs);
359 
360  /* Sedimentation... */
361  if (ctl.qnt_r >= 0 && ctl.qnt_rho >= 0)
362  module_sedi(&ctl, met0, met1, atm, dt);
363 
364  /* Isosurface... */
365  if (ctl.isosurf >= 1 && ctl.isosurf <= 4)
366  module_isosurf(&ctl, met0, met1, atm, cache);
367 
368  /* Check final positions... */
369  module_position(met0, met1, atm, dt);
370 
371  /* Interpolate meteorological data... */
372  if (ctl.met_dt_out > 0
373  && (ctl.met_dt_out < ctl.dt_mod || fmod(t, ctl.met_dt_out) == 0))
374  module_meteo(&ctl, met0, met1, atm);
375 
376  /* Decay of particle mass... */
377  if (ctl.tdec_trop > 0 && ctl.tdec_strat > 0)
378  module_decay(&ctl, atm, dt);
379 
380  /* OH chemistry... */
381  if (ctl.oh_chem[0] > 0 && ctl.oh_chem[2] > 0)
382  module_oh_chem(&ctl, met0, met1, atm, dt);
383 
384  /* Dry deposition... */
385  if (ctl.dry_depo[0] > 0)
386  module_dry_deposition(&ctl, met0, met1, atm, dt);
387 
388  /* Wet deposition... */
389  if ((ctl.wet_depo[0] > 0 || ctl.wet_depo[2] > 0)
390  && (ctl.wet_depo[4] > 0 || ctl.wet_depo[6] > 0))
391  module_wet_deposition(&ctl, met0, met1, atm, dt);
392 
393  /* Boundary conditions... */
394  if (ctl.bound_mass > 0 || ctl.bound_vmr > 0)
395  module_bound_cond(&ctl, met0, met1, atm, dt);
396 
397  /* Write output... */
398  write_output(dirname, &ctl, met0, met1, atm, t);
399  }
400 
401  /* ------------------------------------------------------------
402  Finalize model run...
403  ------------------------------------------------------------ */
404 
405  /* Report problem size... */
406  printf("SIZE_NP = %d\n", atm->np);
407  printf("SIZE_MPI_TASKS = %d\n", size);
408  printf("SIZE_OMP_THREADS = %d\n", omp_get_max_threads());
409  printf("SIZE_ACC_DEVICES = %d\n", num_devices);
410 
411  /* Report memory usage... */
412  printf("MEMORY_ATM = %g MByte\n", sizeof(atm_t) / 1024. / 1024.);
413  printf("MEMORY_CACHE = %g MByte\n", sizeof(cache_t) / 1024. / 1024.);
414  printf("MEMORY_METEO = %g MByte\n", 2 * sizeof(met_t) / 1024. / 1024.);
415  printf("MEMORY_DYNAMIC = %g MByte\n", (sizeof(met_t)
416  + 4 * NP * sizeof(double)
417  + EX * EY * EP * sizeof(float)) /
418  1024. / 1024.);
419  printf("MEMORY_STATIC = %g MByte\n", (EX * EY * sizeof(double)
420  + EX * EY * EP * sizeof(float)
421  + 4 * GX * GY * GZ * sizeof(double)
422  + 2 * GX * GY * GZ * sizeof(int)
423  + 2 * GX * GY * sizeof(double)
424  + GX * GY * sizeof(int)) / 1024. /
425  1024.);
426 
427  /* Delete data region on GPUs... */
428 #ifdef _OPENACC
429  SELECT_TIMER("DELETE_DATA_REGION", NVTX_GPU);
430 #pragma acc exit data delete(ctl,atm,cache,met0,met1,dt,rs)
431 #endif
432 
433  /* Free... */
434  SELECT_TIMER("FREE", NVTX_CPU);
435  free(atm);
436  free(cache);
437  free(met0);
438  free(met1);
439  free(dt);
440  free(rs);
441 
442  /* Report timers... */
443  PRINT_TIMERS;
444  }
445 
446  /* Finalize MPI... */
447 #ifdef MPI
448  MPI_Finalize();
449 #endif
450 
451  /* Stop timers... */
452  STOP_TIMERS;
453 
454  return EXIT_SUCCESS;
455 }

459 void module_advection(
460  met_t * met0,
461  met_t * met1,
462  atm_t * atm,
463  double *dt) {
464 
465  /* Set timer... */
466  SELECT_TIMER("MODULE_ADVECTION", NVTX_GPU);
467 
468 #ifdef _OPENACC
469 #pragma acc data present(met0,met1,atm,dt)
470 #pragma acc parallel loop independent gang vector
471 #else
472 #pragma omp parallel for default(shared)
473 #endif
474  for (int ip = 0; ip < atm->np; ip++)
475  if (dt[ip] != 0) {
476 
477  double u, v, w;
478 
479  /* Interpolate meteorological data... */
480  INTPOL_INIT;
481  INTPOL_3D(u, 1);
482  INTPOL_3D(v, 0);
483  INTPOL_3D(w, 0);
484 
485  /* Get position of the mid point... */
486  double dtm = atm->time[ip] + 0.5 * dt[ip];
487  double xm0 =
488  atm->lon[ip] + DX2DEG(0.5 * dt[ip] * u / 1000., atm->lat[ip]);
489  double xm1 = atm->lat[ip] + DY2DEG(0.5 * dt[ip] * v / 1000.);
490  double xm2 = atm->p[ip] + 0.5 * dt[ip] * w;
491 
492  /* Interpolate meteorological data for mid point... */
493  intpol_met_time_3d(met0, met0->u, met1, met1->u,
494  dtm, xm2, xm0, xm1, &u, ci, cw, 1);
495  intpol_met_time_3d(met0, met0->v, met1, met1->v,
496  dtm, xm2, xm0, xm1, &v, ci, cw, 0);
497  intpol_met_time_3d(met0, met0->w, met1, met1->w,
498  dtm, xm2, xm0, xm1, &w, ci, cw, 0);
499 
500  /* Save new position... */
501  atm->time[ip] += dt[ip];
502  atm->lon[ip] += DX2DEG(dt[ip] * u / 1000., xm1);
503  atm->lat[ip] += DY2DEG(dt[ip] * v / 1000.);
504  atm->p[ip] += dt[ip] * w;
505  }
506 }

510 void module_bound_cond(
511  ctl_t * ctl,
512  met_t * met0,
513  met_t * met1,
514  atm_t * atm,
515  double *dt) {
516 
517  /* Set timer... */
518  SELECT_TIMER("MODULE_BOUNDCOND", NVTX_GPU);
519 
520  /* Check quantity flags... */
521  if (ctl->qnt_m < 0 && ctl->qnt_vmr < 0)
522  ERRMSG("Module needs quantity mass or volume mixing ratio!");
523 
524 #ifdef _OPENACC
525 #pragma acc data present(ctl,met0,met1,atm,dt)
526 #pragma acc parallel loop independent gang vector
527 #else
528 #pragma omp parallel for default(shared)
529 #endif
530  for (int ip = 0; ip < atm->np; ip++)
531  if (dt[ip] != 0) {
532 
533  double ps;
534 
535  /* Check latitude and pressure range... */
536  if (atm->lat[ip] < ctl->bound_lat0 || atm->lat[ip] > ctl->bound_lat1
537  || atm->p[ip] > ctl->bound_p0 || atm->p[ip] < ctl->bound_p1)
538  continue;
539 
540  /* Check surface layer... */
541  if (ctl->bound_dps > 0) {
542 
543  /* Get surface pressure... */
544  INTPOL_INIT;
545  INTPOL_2D(ps, 1);
546 
547  /* Check whether particle is above the surface layer... */
548  if (atm->p[ip] < ps - ctl->bound_dps)
549  continue;
550  }
551 
552  /* Set mass and volume mixing ratio... */
553  if (ctl->qnt_m >= 0)
554  atm->q[ctl->qnt_m][ip] = ctl->bound_mass;
555  if (ctl->qnt_vmr >= 0)
556  atm->q[ctl->qnt_vmr][ip] = ctl->bound_vmr;
557  }
558 }

562 void module_convection(
563  ctl_t * ctl,
564  met_t * met0,
565  met_t * met1,
566  atm_t * atm,
567  double *dt,
568  double *rs) {
569 
570  /* Set timer... */
571  SELECT_TIMER("MODULE_CONVECTION", NVTX_GPU);
572 
573  /* Create random numbers... */
574  module_rng(rs, (size_t) atm->np, 0);
575 
576 #ifdef _OPENACC
577 #pragma acc data present(ctl,met0,met1,atm,dt,rs)
578 #pragma acc parallel loop independent gang vector
579 #else
580 #pragma omp parallel for default(shared)
581 #endif
582  for (int ip = 0; ip < atm->np; ip++)
583  if (dt[ip] != 0) {
584 
585  double cape, pbot, pel, ps, ptop;
586 
587  /* Interpolate CAPE... */
588  INTPOL_INIT;
589  INTPOL_2D(cape, 1);
590 
591  /* Check threshold... */
592  if (isfinite(cape) && cape >= ctl->conv_cape) {
593 
594  /* Interpolate equilibrium level... */
595  INTPOL_2D(pel, 0);
596 
597  /* Check whether particle is above cloud top... */
598  if (!isfinite(pel) || atm->p[ip] < pel)
599  continue;
600 
601  /* Set pressure range for mixing... */
602  pbot = ptop = atm->p[ip];
603  if (ctl->conv_mix_bot == 1) {
604  INTPOL_2D(ps, 0);
605  pbot = ps;
606  }
607  if (ctl->conv_mix_top == 1)
608  ptop = pel;
609 
610  /* Mix in entire in cloud column... */
611  atm->p[ip] = pbot + (ptop - pbot) * rs[ip];
612  }
613  }
614 }

618 void module_decay(
619  ctl_t * ctl,
620  atm_t * atm,
621  double *dt) {
622 
623  /* Set timer... */
624  SELECT_TIMER("MODULE_DECAY", NVTX_GPU);
625 
626  /* Check quantity flags... */
627  if (ctl->qnt_m < 0 && ctl->qnt_vmr < 0)
628  ERRMSG("Module needs quantity mass or volume mixing ratio!");
629 
630 #ifdef _OPENACC
631 #pragma acc data present(ctl,atm,dt)
632 #pragma acc parallel loop independent gang vector
633 #else
634 #pragma omp parallel for default(shared)
635 #endif
636  for (int ip = 0; ip < atm->np; ip++)
637  if (dt[ip] != 0) {
638 
639  /* Get tropopause pressure... */
640  double pt = clim_tropo(atm->time[ip], atm->lat[ip]);
641 
642  /* Get weighting factor... */
643  double w;
644  double p1 = pt * 0.866877899;
645  double p0 = pt / 0.866877899;
646  if (atm->p[ip] > p0)
647  w = 1;
648  else if (atm->p[ip] < p1)
649  w = 0;
650  else
651  w = LIN(p0, 1.0, p1, 0.0, atm->p[ip]);
652 
653  /* Set lifetime... */
654  double tdec = w * ctl->tdec_trop + (1 - w) * ctl->tdec_strat;
655 
656  /* Calculate exponential decay... */
657  double aux = exp(-dt[ip] / tdec);
658  if (ctl->qnt_m >= 0)
659  atm->q[ctl->qnt_m][ip] *= aux;
660  if (ctl->qnt_vmr >= 0)
661  atm->q[ctl->qnt_vmr][ip] *= aux;
662  }
663 }

667 void module_diffusion_meso(
668  ctl_t * ctl,
669  met_t * met0,
670  met_t * met1,
671  atm_t * atm,
672  cache_t * cache,
673  double *dt,
674  double *rs) {
675 
676  /* Set timer... */
677  SELECT_TIMER("MODULE_TURBMESO", NVTX_GPU);
678 
679  /* Create random numbers... */
680  module_rng(rs, 3 * (size_t) atm->np, 1);
681 
682 #ifdef _OPENACC
683 #pragma acc data present(ctl,met0,met1,atm,cache,dt,rs)
684 #pragma acc parallel loop independent gang vector
685 #else
686 #pragma omp parallel for default(shared)
687 #endif
688  for (int ip = 0; ip < atm->np; ip++)
689  if (dt[ip] != 0) {
690 
691  double u[16], v[16], w[16];
692 
693  /* Get indices... */
694  int ix = locate_reg(met0->lon, met0->nx, atm->lon[ip]);
695  int iy = locate_reg(met0->lat, met0->ny, atm->lat[ip]);
696  int iz = locate_irr(met0->p, met0->np, atm->p[ip]);
